P(B) = 0.35

P(d|A) = 0.005
P(d|B) = 0.008
P(d|C) = 0.001

P(d) = 0.25*0.005 + 0.35*0.008 + 0.40*0.001 = 0.00445

P(A|d) = P(d|A)P(A)/P(d) = 0.005*0.25/0.00445 = 0.281
P(B|d) = P(d|B)P(B)/P(d) = 0.008*0.35/0.00445 = 0.629
P(C|d) = P(d|C)P(C)/P(d) = 0.001*0.40/0.00445 = 0.090
